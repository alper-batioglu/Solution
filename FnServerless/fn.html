Init fn project

fn init --runtime node <Function name>

List fn contexts (for local fndemouser)

fn list contexts

Create App

fn create app <App Name>

List Apps

fn list apps

Deploy Function to App(create docker image and run container)

fn --verbose deploy --app <App Name> --local

List Functions deployed

fn list functions <App Name>

Invoke deployed Function

fn invoke <App Name> <Function name>

with params

echo -n '<JSON Data>' | fn invoke <App Name> <Function name> --content-type application/json

with curl

curl -X "POST" -H "Content-Type: application/json" -d '<JSON Data>' <Function URL>

get Function URL

fn inspect function <App Name> <Function name>

delete Function

fn delete function <App Name> <Function name>

delete App

fn delete app <App Name>
